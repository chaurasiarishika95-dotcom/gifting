/* ── GEMINI API (SECURE VIA BACKEND ONLY) ── */ async function ask(prompt, max = 1200) { try { const response = await fetch("/api/suggest", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ prompt, max_tokens: max }) }); if (!response.ok) { throw new Error(`API Error: ${response.status}`); } const data = await response.json(); return data.text || ""; } catch (error) { console.error("API error:", error); return "Something went wrong. Please try again."; } }
